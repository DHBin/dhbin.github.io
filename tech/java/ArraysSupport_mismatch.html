                                                              elasticsearchжҠҠеӯ—з¬ҰдёІзұ»еһӢз»ҹдёҖеӯҳеӮЁдёәbyteж•°з»„пјҢжүҖд»Ҙеӯ—з¬ҰдёІзҡ„еҜ№жҜ”е®һйҷ…дёҠе°ұжҳҜbyteж•°з»„зҡ„еҜ№жҜ”гҖӮеҲ°иҝҷйҮҢпјҢжҲ‘дә§з”ҹдәҶдёҖдёӘз–‘й—®пјҢesзҡ„еә•еұӮжҳҜжҖҺд№ҲеҜ№жҜ”byteж•°з»„зҡ„пјҹиҝҷе…ізі»еҲ°еӯ—з¬ҰдёІжҺ’еәҸзҡ„жҖ§иғҪгҖӮи„‘еӯҗйҮҢзҡ„зӯ”жЎҲе°ұжҳҜдёҖдёӘforеҫӘзҺҜйҒҚеҺҶдёӨдёӘж•°з»„дёҖдёҖеҜ№жҜ”пјҢжңҖеҗҺеҸ‘зҺ°еңЁJDK8зҡ„зЎ®жҳҜиҝҷд№Ҳе®һзҺ°зҡ„пјҢдҪҶжҳҜesжңҖдҪҺзүҲжң¬иҰҒжұӮе·Із»ҸжҳҜjdk11пјҢjdk9зҡ„ж—¶еҖҷе°ұеҜ№ж•°з»„зҡ„еҜ№жҜ”иҝӣиЎҢдәҶдјҳеҢ–гҖӮжәҗз ҒеҰӮдёӢпјҡ

                                                                  public static int compare(byte[] a, byte[] b) {
        if (a == b)
            return 0;
        if (a == null || b == null)
            return a == null ? -1 : 1;

        // е…ій”®зҡ„д»Јз ҒеңЁиҝҷйҮҢ
        int i = ArraysSupport.mismatch(a, b,
                                       Math.min(a.length, b.length));
        if (i >= 0) {
            return Byte.compare(a[i], b[i]);
        }

        return a.length - b.length;
    }
    public static int mismatch(byte[] a,
                               byte[] b,
                               int length) {
        // ISSUE: defer to index receiving methods if performance is good
        // assert length <= a.length
        // assert length <= b.length

        int i = 0;
        // еӣ дёәlongзұ»еһӢжҳҜ8byte
        if (length > 7) {
            if (a[0] != b[0])
                return 0;
            // е…ій”®д»Јз ҒеңЁиҝҷ
            i = vectorizedMismatch(
                    a, Unsafe.ARRAY_BYTE_BASE_OFFSET,
                    b, Unsafe.ARRAY_BYTE_BASE_OFFSET,
                    length, LOG2_ARRAY_BYTE_INDEX_SCALE);
            if (i >= 0)
                return i;
            // Align to tail
            i = length - ~i;
//            assert i >= 0 && i <= 7;
        }
        // Tail < 8 bytes
        for (; i < length; i++) {
            if (a[i] != b[i])
                return i;
        }
        return -1;
    }
    public static int vectorizedMismatch(Object a, long aOffset,
                                         Object b, long bOffset,
                                         int length,
                                         int log2ArrayIndexScale) {
        // assert a.getClass().isArray();
        // assert b.getClass().isArray();
        // assert 0 <= length <= sizeOf(a)
        // assert 0 <= length <= sizeOf(b)
        // assert 0 <= log2ArrayIndexScale <= 3

        int log2ValuesPerWidth = LOG2_ARRAY_LONG_INDEX_SCALE - log2ArrayIndexScale;
        int wi = 0;
        for (; wi < length >> log2ValuesPerWidth; wi++) {
            long bi = ((long) wi) << LOG2_ARRAY_LONG_INDEX_SCALE;
            long av = U.getLongUnaligned(a, aOffset + bi);
            long bv = U.getLongUnaligned(b, bOffset + bi);
            if (av != bv) {
                long x = av ^ bv;
                int o = BIG_ENDIAN
                        ? Long.numberOfLeadingZeros(x) >> (LOG2_BYTE_BIT_SIZE + log2ArrayIndexScale)
                        : Long.numberOfTrailingZeros(x) >> (LOG2_BYTE_BIT_SIZE + log2ArrayIndexScale);
                return (wi << log2ValuesPerWidth) + o;
            }
        }

                                                              жғ…еҶөдәҢпјҡж•°з»„й•ҝеәҰеӨ§дәҺзӯүдәҺ8

                                                              жҲ‘д»¬зҹҘйҒ“ж•°з»„еңЁеҶ…еӯҳдёӯжҳҜд»ҘдёҖеқ—иҝһз»ӯзҡ„еҶ…еӯҳеӯҳеӮЁзҡ„пјҢиҝҷж ·е°ұеҸҜд»ҘжҠҠ8bytesж•°жҚ®иҪ¬жҲҗlongзұ»еһӢжқҘеҜ№жҜ”гҖӮеҒҮи®ҫдёҖдёӘж•°з»„зҡ„й•ҝеәҰжҳҜ24bytesпјҢжүҖд»Ҙjdk8зҡ„ж–№жі•ж—¶й—ҙеӨҚжқӮеәҰжҳҜO(24)пјҢjdk9зҡ„ж–№жі•жҳҜO(24 / 8) = O(3)гҖӮеҮҸе°‘дәҶеҫӘзҺҜзҡ„ж¬Ўж•°гҖӮ

                                                              йӮЈй—®йўҳе°ұжқҘдәҶпјҢjdkжҳҜеҰӮдҪ•жҠҠbyte[]иҪ¬longзұ»еһӢзҡ„пјҢеңЁеёёи§„зҡ„ејҖеҸ‘жІЎжңүиҝҷдёӘж“ҚдҪңе‘ҖпјҢеҰӮжһңжҳҜforеҫӘзҺҜ8ж¬ЎеҶҚиҪ¬longпјҢиҝҷеӨҚжқӮеәҰд№ҹжІЎжңүеҮҸдҪҺе‘ҖпјҢзЎ®е®һпјҢjdkйҮҮз”ЁдәҶжӣҙйӘҡзҡ„ж–№жі•гҖӮUnsafeзұ»пјҢиҝҷдёӘзұ»еҰӮе…¶еҗҚпјҢжҳҜдёҚе®үе…Ёзҡ„пјҢиғҪеӨҹеғҸC/C++иҜӯиЁҖдёҖж ·ж“ҚдҪңеҶ…еӯҳпјҢдёҠйқўдҪҝз”ЁеҲ°зҡ„е…ій”®APIжҳҜUnsafe#getLongUnaligned(java.lang.Object, long)пјҢеҸӮж•°1дј е…Ҙж•°з»„пјҢеҸӮж•°2дј е…ҘеҒҸз§»йҮҸ

public class LongOpt {

    public static void main(String[] args) {
        Unsafe unsafe = UnsafeUtil.UNSAFE;

        byte[] bytes = "0200000001000000".getBytes(StandardCharsets.UTF_8);
        // Unsafe.ARRAY_BYTE_BASE_OFFSETжҳҜж•°з»„еӨҙйғЁж•°жҚ®еӨ§е°ҸпјҢжҠҠиҝҷдёӘеҠ дёҠе°ұжҳҜж•°з»„е…ғзҙ зҡ„еҲқе§ӢдҪҚзҪ®
        long l1 = unsafe.getLong(bytes, Unsafe.ARRAY_BYTE_BASE_OFFSET);
        // +8зҡ„еҺҹеӣ жҳҜlongзҡ„й•ҝеәҰжҳҜ8дёӘеӯ—иҠӮ
        long l2 = unsafe.getLong(bytes, Unsafe.ARRAY_BYTE_BASE_OFFSET + 8);
        System.out.println(l1); // 3472328296227680816
        System.out.println(l2); // 3472328296227680560
    }


}
